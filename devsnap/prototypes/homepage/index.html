useDebounce.ts 
import { useState, useEffect } from 'react'

export function useDebounce<T>(
  value: T,
  delay: number
): T {
  const [debounced, setDebounced] =
    useState<T>(value)

  useEffect(() => {
    const t = setTimeout(() =>
      setDebounced(value), delay)
    return () => clearTimeout(t)
  }, [value, delay])

  return debounced
}
